        def bubble(title, text, options={:timeout => 3, :alpha => 0.9, :x => "right", :y => "top", :icon => nil})
                %x{
                        #{CD} bubble --title #{title} \
                        --text #{text} \
                        --timeout #{options[:timeout]} \
                        --x-placement #{options[:x]} \
                        --y-placement #{options[:y]} \
                        --icon #{options[:icon]}
                }
        end
